from flask import Flask, request, jsonify
import requests
app = Flask(__name__)

@app.route("/login/<user>/<password>", methods=["GET"])
def test_login(user, password):
    res = requests.post(
        "https://roble-api.openlab.uninorte.edu.co/auth/probando_49357d021e/login",
        json={
            "email": user,
            "password": password
        }
    )
    return res.json()

if __name__ == "__main__":
    app.run(host="0.0.0.0", port=8000)
