from flask import Flask, jsonify

app = Flask(__name__)

@app.route("/saludo", methods=["GET"])
def saludo():
    return jsonify({"mensaje": "Hola, este es un microservicio sencillo!"})

@app.route("/doble/<int:num>", methods=["GET"])
def doble(num):
    return jsonify({"numero": num, "doble": num * 2})

if __name__ == "__main__":
    app.run(host="0.0.0.0", port=8000)
